A list of tabs.

## Installation

```bash
npm i @salad-ui/tabs
```

## Usage

```jsx
import {Tabs} from '@salad-ui/tabs';

<Tabs>
  <Tabs.Tab role="link" href="#plugins/featured">
    Featured
  </Tabs.Tab>
  <Tabs.Tab role="link" href="#plugins/popular">
    Popular
  </Tabs.Tab>
  <Tabs.Tab role="link" href="#plugins/recommended">
    Recommended
  </Tabs.Tab>
  <Tabs.Tab role="link" href="#plugins/favorites">
    Favorites
  </Tabs.Tab>
</Tabs>;
```
